// Define errors once — your domain's single source of truth
public static class UserErrors
{
    public static Error NotFound(Guid id) =>
        Error.NotFound("User.NotFound", $"User '{id}' not found.");

    public static readonly Error InvalidEmail =
        Error.Validation("User.InvalidEmail"); // message defaults to code
}

// Return results, never throw
public async Task<Result<User>> GetByIdAsync(Guid id)
{
    var user = await _repo.FindAsync(id);
    return user is null ? UserErrors.NotFound(id) : Result.Success(user);
}

// Consume with Match — no try/catch, no null checks
return result.Match(
    onSuccess: user  => Ok(user),
    onFailure: error => error.Type switch
    {
        ErrorType.NotFound   => NotFound(),
        ErrorType.Validation => BadRequest(error.Message),
        _                    => Problem(error.Message)
    });

Philosophy

Built on three principles.

Every decision in Arkn follows the same north star.

  • ✦
    Zero lock-in
    Every package is independent. No package forces you to use another. Swap or remove anything at any time.
  • ✦
    Explicit over magic
    No hidden behaviors, no ambient context, no conventions that only work if you read the docs. What you write is what runs.
  • ✦
    Failures are first-class
    Result<T> makes every failure explicit at the type level. No more unexpected exceptions, no more null reference surprises.
  • ✦
    Copilot-ready
    APIs are consistent enough that LLMs generate correct Arkn code on the first attempt. Ships with instructions for Copilot, Cursor and Claude.
  • ✦
    Zero external dependencies in core
    Arkn.Core and Arkn.Results have zero dependencies. Infrastructure packages use only Microsoft.Extensions.*
  • ✦
    152 tests, always green
    Targets .NET 9 and .NET 10. Every package is covered by unit tests. CI runs on ubuntu-latest and windows-latest on every push.

  .Bind(v => next(v))
  .Tap(v => sideEffect(v))
  .Match(
    v => Ok(v),
    e => Problem(e.Message));
Arkn.Copilot

Patterns enforced at compile time.

Arkn.Analyzers ships with four Roslyn rules that catch violations before they reach code review — or production.

ARK001
Domain methods must return Result<T> warning
Detects public domain/service methods that throw exceptions instead of returning Result or Result<T>. Includes a code fix that wraps the return type automatically.
ARK002
Error codes must follow Namespace.Reason warning
Validates that every Error.* call uses a code in the format Namespace.Reason (e.g. User.NotFound). Catches typos like "usernotfound" or "ERROR_404".
ARK003
Result must not be silently discarded warning
Warns when a Result or Result<T> value is discarded without consuming .IsSuccess, .Match(), .Bind() or .Map().
ARK004
IArknJob.ExecuteAsync must return Task<Result> error
Enforces that every IArknJob implementation returns Task<Result> or Task<Result<T>>. Returning plain Task or void breaks the job runner contract.

Also ships with .github/copilot-instructions.md, .cursor/rules/arkn.mdc and CLAUDE.md — so your AI assistant always generates correct Arkn code.

Get Started

Up and running in minutes.

Install individual packages or scaffold a full project with the templates.

dotnet CLI
# Core + Results
dotnet add package Arkn.Results
dotnet add package Arkn.Core

# Infrastructure
dotnet add package Arkn.Http
dotnet add package Arkn.Jobs
dotnet add package Arkn.Logging
dotnet add package Arkn.Notifications

# Analyzers (dev only)
dotnet add package Arkn.Analyzers

# Application Insights sink
dotnet add package Arkn.Extensions.Logging.ApplicationInsights
Templates
# Install template pack
dotnet new install Arkn.Templates

# Scaffold a Minimal API
dotnet new arkn-api -n MyApi

# Scaffold a Worker with Jobs
dotnet new arkn-job -n MyWorker

# Scaffold a Class Library
dotnet new arkn-lib -n MyLibrary
Program.cs — Minimal API with Arkn.Jobs
builder.Services.AddArknJobs(jobs =>
{
    jobs.Add<InvoiceProcessorJob>("0 2 * * *")
        .WithRetry(maxAttempts: 3)
        .WithTimeout(TimeSpan.FromMinutes(10))
        .NotifyOn(JobEvent.Failed | JobEvent.TimedOut);

    jobs.OnFailure<SlackNotifier>();
});
