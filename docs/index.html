// Define errors once — never inline Error.* calls
public static class OrderErrors
{
    public static Error NotFound(Guid id) =>
        Error.NotFound("Order.NotFound", $"Order '{id}' not found.");
    public static readonly Error InvalidAmount =
        Error.Validation("Order.InvalidAmount"); // message defaults to code
}

// Return results, never throw
public async Task<Result<Order>> GetAsync(Guid id)
{
    var order = await _repo.FindAsync(id);
    return order is null ? OrderErrors.NotFound(id) : Result.Success(order);
}

// Consume with Match — exhaustive, compiler-enforced
return result.Match(
    onSuccess: order => Ok(order),
    onFailure: error => error.Type switch
    {
        ErrorType.NotFound   => NotFound(new { error.Code, error.Message }),
        ErrorType.Validation => BadRequest(new { error.Code, error.Message }),
        _                    => Problem(error.Message)
    });

Exceptions as control flow

Implicit contracts

Your service returns User?. Is null "not found", "not loaded", or "access denied"? Nobody knows until runtime.

Framework gravity

.github/copilot-instructions.md 
// .github/copilot-instructions.md

✅ Always return Result<T> from domain methods — never throw
✅ Error codes: Namespace.Reason (User.NotFound, Order.Invalid)
✅ Consume results with .Match() — never access .Value directly
✅ IArknJob.ExecuteAsync must return Task<Result>
ARK001

ErrorGroup required error

Errors must be defined in a static error group class. Inline Error.* calls are not allowed.

ARK002

Error code format error

Error codes must follow Namespace.Reason convention.

ARK003

Result not unwrapped error

Do not access .Value directly. Use .Match() to consume results.

ARK004

Job return type error

IArknJob.ExecuteAsync must return Task<Result>, never Task.

Arkn.Results v—
The foundation. Explicit failures with Result<T>, typed errors, and chainable extensions — no exceptions as control flow.
dotnet add package Arkn.Results
resultserrorsfunctionalzero-deps
Code example
csharp
// message is optional — defaults to code
Error.NotFound("User.NotFound")
Error.Validation("Email.Invalid", "Custom message when needed.")

// Chain operations
Result<string> result = await GetUserAsync(id)
    .MapAsync(u => u.Name)
    .BindAsync(name => ValidateAsync(name));

csharp
public sealed class PaymentsClient(IArknHttp http)
    : ArknHttpClient(http, "https://api.payments.com")
{
    public Task<Result<Payment>> GetAsync(Guid id) => GetAs<Payment>("/payments/{id}", id);
    public Task<Result<Payment>> CreateAsync(PaymentRequest r) => PostAs<Payment>("/payments", r);

// Registration — full stack

    .WithResponseCaching()
    .WithDebugLogging(DebugLoggingOptions.Production);

Arkn.Jobs v—
In-process cron scheduler. No database, no external process. ARK004 enforces Task<Result> return types at compile time.
dotnet add package Arkn.Jobs
jobscronschedulerin-processretry
Code example
csharp
// Job — ARK004 enforces Task<Result> return type
public class InvoiceJob(IArknLogger logger) : IArknJob
{
    public async Task<Result> ExecuteAsync(ArknJobContext ctx)
    {
        logger.Info($"Processing. RunId={ctx.RunId}");
        await ProcessAsync(ctx.CancellationToken);
        return Result.Success();
    }
}

// Registration
services.AddArknJobs(jobs =>
{
    jobs.Add<InvoiceJob>("0 2 * * *")
        .WithRetry(maxAttempts: 3)
        .WithTimeout(TimeSpan.FromMinutes(10))
        .NotifyOn(JobEvent.Failed | JobEvent.TimedOut);
    jobs.OnFailure<SlackNotifier>();
});

Rules reference
csharp — before / after
// ❌ ARK001 — inline Error.* call (not in an ErrorGroup)
return Error.NotFound("User.NotFound");

// ✅ Correct — defined in a static error class
public static class UserErrors
{
    public static Error NotFound(Guid id) => Error.NotFound("User.NotFound", $"User '{id}' not found.");
}

// ❌ ARK003 — accessing .Value without checking
var user = result.Value;

// ✅ Correct — exhaustive match
var user = result.Match(onSuccess: u => u, onFailure: _ => null);

// ❌ ARK004 — job returning Task instead of Task<Result>
public async Task ExecuteAsync(ArknJobContext ctx) { ... }

// ✅ Correct
public async Task<Result> ExecuteAsync(ArknJobContext ctx) { ... }
